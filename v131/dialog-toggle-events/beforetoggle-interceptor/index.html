beforetoggle Interceptor

A multi-step wizard dialog that uses beforetoggle to intercept close attempts mid-flow. Instead of a simple dirty-check alert, the dialog catches the close, slides to an in-dialog confirmation step, and lets the user choose: resume or abandon. Calling event.preventDefault() keeps the dialog open while the internal state changes.

the code

let currentStep = 1;
let showingExitConfirm = false;

dialog.addEventListener("beforetoggle", (e) => {
  if (e.newState !== "closed") return;
  if (!interceptEnabled) return;
  if (currentStep <= 1) return; // step 1 has no data yet

  // Prevent the default close
  e.preventDefault();

  // Instead: show an in-dialog confirmation
  showExitConfirm();
});

function showExitConfirm() {
  hideAllSteps();
  document.getElementById("step-exit").style.display = "";
  showingExitConfirm = true;
}

resumeBtn.onclick = () => {
  showingExitConfirm = false;
  showStep(currentStep);
};

abandonBtn.onclick = () => {
  showingExitConfirm = false;
  interceptEnabled = false; // disable guard for this close
  dialog.close();
  interceptEnabled = true;
};
