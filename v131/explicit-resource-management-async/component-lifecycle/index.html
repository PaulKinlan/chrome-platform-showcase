Component Lifecycle

A custom element that uses await using inside its connected/disconnected cycle to register and automatically clean up an IntersectionObserver, a ResizeObserver, and an interval timer — no manual teardown code, no dangling listeners after the element is removed from the DOM.

the code

class LiveCard extends HTMLElement {
  // Holds the AsyncDisposableStack for this component instance
  #resources = null;

  async connectedCallback() {
    const stack = new AsyncDisposableStack();

    // Register IntersectionObserver as a disposable
    const io = new IntersectionObserver((entries) => {
      this.#onIntersect(entries);
    });
    io.observe(this);
    stack.defer(async () => io.disconnect()); // auto-disconnect on dispose

    // Register ResizeObserver as a disposable
    const ro = new ResizeObserver(() => this.#onResize());
    ro.observe(this);
    stack.defer(async () => ro.disconnect());

    // Register an interval as a disposable
    const id = setInterval(() => this.#onTick(), 1000);
    stack.defer(async () => clearInterval(id));

    this.#resources = stack;
  }

  disconnectedCallback() {
    this.#resources?.disposeAsync(); // cleans up all three at once
    this.#resources = null;
  }
}
