the code

.card {
  border-radius: 40px;
  corner-shape: squircle;          /* round | squircle | scoop | bevel | notch */
}

.card.custom {
  corner-shape: superellipse(2.5); /* k > 2 = squircle, k < 2 = scoop */
}
