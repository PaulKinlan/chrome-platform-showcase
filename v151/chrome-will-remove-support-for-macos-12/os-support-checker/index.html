detection code

const hints = await navigator.userAgentData.getHighEntropyValues([
  'platform',
  'platformVersion',
]);

if (hints.platform === 'macOS') {
  const major = parseInt(hints.platformVersion.split('.')[0], 10);
  const supported = major >= 13; // Chrome 151 minimum: macOS 13
  if (!supported) {
    showUpdateBanner();
  }
}
