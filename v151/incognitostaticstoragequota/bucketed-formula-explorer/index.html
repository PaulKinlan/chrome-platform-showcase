code (the fix)

// Chromium pseudo-code: pre-Chrome 151 incognito branch.
function quotaIncognito_v0(ramBytes, usageBytes) {
  // Bucketed only against the RAM-derived raw value.
  const raw = ramBytes / 20;
  return Math.ceil(raw / 1e9) * 1e9 + usageBytes;
}

// Chromium pseudo-code: Chrome 151+ unified branch.
function quota_v151(diskBytes, usageBytes) {
  // Apply the same disk-derived bucketing regardless of mode.
  const raw = Math.min(diskBytes, 10n * 2n ** 30n);
  return Math.max(10n * 2n ** 30n, BigInt(Math.ceil(Number(raw) / 1e9)) * 10n ** 9n) + BigInt(usageBytes);
}
