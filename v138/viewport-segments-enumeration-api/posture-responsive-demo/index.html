the detection code

const segments = window.getWindowSegments?.() ?? [];

if (segments.length === 2) {
  const [left, right] = segments;
  const isBookFold = left.top === right.top;    // side by side
  const isTentFold = left.left === right.left;  // stacked

  if (isBookFold) {
    // Show sidebar in left pane, content in right pane
    layout.setAttribute('data-fold', 'book');
  } else if (isTentFold) {
    // Show content above fold, controls below fold
    layout.setAttribute('data-fold', 'tent');
  }
}

// CSS counterpart:
@media (spanning: single-fold-vertical) {
  .layout { grid-template-columns:
    env(viewport-segment-width 0 0)
    env(viewport-segment-width 1 0); }
}
