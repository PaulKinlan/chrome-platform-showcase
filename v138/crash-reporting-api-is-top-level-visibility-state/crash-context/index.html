the code

const observer = new ReportingObserver((reports) => {
  for (const r of reports) {
    if (r.type !== "crash") continue;
    console.log({
      reason: r.body.reason,
      isTopLevel: r.body.isTopLevel,           // NEW in 138
      visibilityState: r.body.visibilityState, // NEW in 138
    });
  }
});
observer.observe();
