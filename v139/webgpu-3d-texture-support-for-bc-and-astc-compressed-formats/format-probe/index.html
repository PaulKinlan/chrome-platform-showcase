the code

const adapter = await navigator.gpu.requestAdapter();
console.log(adapter.features.has("texture-compression-bc"));        // 2D BC (existing)
console.log(adapter.features.has("texture-compression-bc-sliced-3d")); // 3D BC (NEW)
console.log(adapter.features.has("texture-compression-astc-sliced-3d")); // 3D ASTC (NEW)

const device = await adapter.requestDevice({
  requiredFeatures: ["texture-compression-bc-sliced-3d"],
});
