the code

.ticket {
  border-radius: 24px;
  corner-shape: scoop;  /* concave arc, like a hole-punched stub */
}

.stamp {
  border-radius: 18px;
  corner-shape: notch;  /* concave V cut */
}

.button-soft {
  border-radius: 16px;
  corner-shape: squircle;  /* iOS-style smooth corner */
}
