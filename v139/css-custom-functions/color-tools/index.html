Color Tools via CSS Custom Functions

Author-defined @function rules that wrap oklch(from …) relative color syntax for lighten, darken, saturate, alpha, and mix operations. One function call per operation — no preprocessor, no JS, pure CSS.

/* Declare once, use anywhere — no preprocessor needed */
@function --lighten(--base <color>, --amount <percentage>) {
  result: oklch(from var(--base) calc(l + var(--amount) / 100) c h);
}
@function --darken(--base <color>, --amount <percentage>) {
  result: oklch(from var(--base) calc(l - var(--amount) / 100) c h);
}
@function --alpha(--base <color>, --opacity <number>) {
  result: oklch(from var(--base) l c h / var(--opacity));
}
@function --mix-ratio(--a <color>, --b <color>, --ratio <percentage>) {
  result: color-mix(in oklch, var(--a) calc(100% - var(--ratio)), var(--b) var(--ratio));
}

/* Usage */
.btn-hover { background: --lighten(var(--brand), 15%); }
.btn-press { background: --darken(var(--brand), 10%); }
.overlay   { background: --alpha(var(--brand), 0.3); }
.gradient-mid { background: --mix-ratio(var(--start), var(--end), 50%); }
