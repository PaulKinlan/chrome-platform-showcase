runtime check

// per-language probe
const status = await SpeechRecognition.availableOnDevice('en-US');
// → 'available' | 'downloadable' | 'unavailable'

if (status === 'downloadable') {
  // user gesture required; this kicks off a model download
  await SpeechRecognition.installOnDevice('en-US');
}

// configure the recogniser
const recogniser = new SpeechRecognition();
recogniser.lang = 'en-US';
recogniser.processLocally = true;       // 139: requests the local pipeline
recogniser.continuous = true;
recogniser.onresult = (e) => render(e.results);
recogniser.start();
