the code

// Detect page lifecycle support
const supported = 'wasDiscarded' in document;

document.addEventListener("visibilitychange", () => {
  if (document.visibilityState === "hidden") {
    // v139: you have ~30 s before Android freeze
    // pre-v139: you had ~5 minutes
    flushAnalytics();
  }
});

document.addEventListener("freeze", () => {
  // last chance before CPU is pulled
  persistCriticalState();
});

// On resume, check if we were discarded
if (document.wasDiscarded) {
  restoreFromStorage();
}
