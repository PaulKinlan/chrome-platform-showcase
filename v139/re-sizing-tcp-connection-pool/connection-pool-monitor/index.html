Resource Timing API — measure connection wait

const [entry] = performance.getEntriesByName(url, 'resource');
const connectionWait = entry.connectEnd - entry.connectStart;
// connectStart: moment socket allocation began (waiting for pool slot if pool full)
// connectEnd:   moment TCP handshake completed
// if connectStart === connectEnd → connection was reused from idle pool
console.log('TCP connect overhead:', connectionWait.toFixed(1), 'ms');
