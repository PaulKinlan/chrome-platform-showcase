ed.addEventListener('beforeinput', (e) => {
  const distinct = /^delete(Content|ByCut|HardLineBackward|ByDrag)$/.test(e.inputType);
  if (distinct) flush(); // start a new undo step
  else stay();           // coalesce with previous backspace
});
