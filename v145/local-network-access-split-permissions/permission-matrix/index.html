code

// Chrome 145 splits the LNA permission into two prompts:
//   "local-network" for private/loopback targets from a public origin
//   "local-network-loopback" specifically for 127.0.0.0/8 + ::1
// Use Permissions API to query state without prompting.
const lna = await navigator.permissions.query({ name: 'local-network' });
const lb  = await navigator.permissions.query({ name: 'local-network-loopback' });
