Fallback pattern

/* Detect element-scoped startViewTransition */ const SUPPORTS_ELEMENT_VT = typeof Element.prototype.startViewTransition === 'function'; /* Wrapper that degrades gracefully */ function transitionElement(el, updateFn) { if (SUPPORTS_ELEMENT_VT) { // Chrome 147+: scoped to this element, concurrent transitions possible return el.startViewTransition(updateFn); } if (typeof document.startViewTransition === 'function') { // Chrome 111+: global view transition (no concurrent support) return document.startViewTransition(updateFn); } // Fallback: instant update updateFn(); return Promise.resolve(); } /* Usage */ const panel = document.querySelector('.panel'); transitionElement(panel, () => { panel.innerHTML = newContent; });
