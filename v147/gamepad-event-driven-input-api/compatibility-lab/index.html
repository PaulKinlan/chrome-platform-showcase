Input method comparison

rawgamepadinputchange CHROME 147 OT
Push-based. Fires on every hardware input change at device rate (240Hz+).
Sub-frame fidelity. No wasted JS. Low latency.
Event: rawgamepadinputchange
Also: gamepadbuttondown/up, gamepadaxismove
rAF polling ALL BROWSERS
Pull-based. Reads navigator.getGamepads() each frame (60Hz).
Input quantised to render cadence. Misses sub-frame events.
Works everywhere without flags.

Fallback pattern

/* Detect gamepad event-driven API */ const HAS_RAW_INPUT = 'ongamepadinputchange' in window || 'onrawgamepadinputchange' in window; const HAS_BTN_EVENTS = 'ongamepadbuttondown' in window; const HAS_AXIS_EVENTS = 'ongamepadaxismove' in window; const HAS_GETGAMEPADS = typeof navigator.getGamepads === 'function'; /* Universal gamepad input handler */ function setupGamepadInput(callbacks) { const { onButtonDown, onButtonUp, onAxisMove } = callbacks; if (HAS_BTN_EVENTS) { // Chrome 147 OT — push-based, sub-frame fidelity window.addEventListener('gamepadbuttondown', (e) => onButtonDown?.(e.gamepad, e.button)); window.addEventListener('gamepadbuttonup', (e) => onButtonUp?.(e.gamepad, e.button)); window.addEventListener('gamepadaxismove', (e) => onAxisMove?.(e.gamepad, e.axis, e.value)); return; } if (!HAS_GETGAMEPADS) return; // Fallback: rAF polling let prevState = {}; function poll() { const gamepads = navigator.getGamepads(); for (const gp of gamepads) { if (!gp) continue; const prev = prevState[gp.index] ?? { buttons: [], axes: [] }; gp.buttons.forEach((btn, i) => { if (btn.pressed && !prev.buttons[i]?.pressed) onButtonDown?.(gp, i); if (!btn.pressed && prev.buttons[i]?.pressed) onButtonUp?.(gp, i); }); gp.axes.forEach((val, i) => { if (Math.abs(val - (prev.axes[i] ?? 0)) > 0.01) onAxisMove?.(gp, i, val); }); prevState[gp.index] = { buttons: gp.buttons.map(b => ({ pressed: b.pressed, value: b.value })), axes: [...gp.axes], }; } requestAnimationFrame(poll); } requestAnimationFrame(poll); }
