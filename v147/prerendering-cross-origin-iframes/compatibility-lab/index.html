Opt-in header pattern

/* Detect prerendering state in the current page */ const IS_PRERENDERING = document.prerendering === true; const HAS_PRERENDERING_API = 'prerendering' in document; /* Listen for activation (when prerendered page becomes active) */ if (IS_PRERENDERING) { document.addEventListener('prerenderingchange', () => { // Page is now active — run deferred initialisation initAnalytics(); startVideoAutoplay(); }); } /* Check activation timing */ function getPrerenderActivationTime() { const nav = performance.getEntriesByType('navigation')[0]; if (nav && 'activationStart' in nav) { // activationStart > 0 means this page was prerendered return nav.activationStart; } return 0; } /* For cross-origin iframes to be prerendered, the iframe's server must respond with this header: Supports-Loading-Mode: prerender-cross-origin-frames Without it, the iframe is skipped during prerender and loaded normally when the page activates. Server example (Node.js): res.set('Supports-Loading-Mode', 'prerender-cross-origin-frames'); */ /* Detect if loading in prerendered context from inside an iframe */ function isInPrerenderContext() { try { // If parent is same origin, check parent's prerendering state return window.parent !== window && window.parent.document.prerendering; } catch { // Cross-origin parent — can't read prerendering state return false; } }
