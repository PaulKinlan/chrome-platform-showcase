/* Detect inline path attribute support */ function supportsTextPathInlinePath() { const svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg'); const tp = document.createElementNS('http://www.w3.org/2000/svg', 'textPath'); tp.setAttribute('path', 'M0 0 L100 0'); svg.appendChild(tp); document.body.appendChild(svg); const ok = tp.getAttribute('path') === 'M0 0 L100 0' && typeof tp.getAttribute('path') === 'string'; document.body.removeChild(svg); return ok; } const HAS_INLINE_PATH = supportsTextPathInlinePath(); /* Fallback: always works — use defs + href */ function makeTextPath(text, pathD) { const svgNS = 'http://www.w3.org/2000/svg'; const svg = document.createElementNS(svgNS, 'svg'); if (HAS_INLINE_PATH) { // Chrome 147+: no <defs> needed const textEl = document.createElementNS(svgNS, 'text'); const tp = document.createElementNS(svgNS, 'textPath'); tp.setAttribute('path', pathD); tp.textContent = text; textEl.appendChild(tp); svg.appendChild(textEl); } else { // Fallback: define path in <defs>, reference via href const defs = document.createElementNS(svgNS, 'defs'); const path = document.createElementNS(svgNS, 'path'); const id = 'tp-' + Math.random().toString(36).slice(2); path.setAttribute('id', id); path.setAttribute('d', pathD); defs.appendChild(path); svg.appendChild(defs); const textEl = document.createElementNS(svgNS, 'text'); const tp = document.createElementNS(svgNS, 'textPath'); tp.setAttribute('href', '#' + id); tp.textContent = text; textEl.appendChild(tp); svg.appendChild(textEl); } return svg; }
