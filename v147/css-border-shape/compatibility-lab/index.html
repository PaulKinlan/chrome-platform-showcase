Fallback pattern

/* border-shape not supported — fallback via clip-path */ @supports not (border-shape: shape(from 0% 0%, line to 100% 0%, close)) { .shaped-el { /* clip-path shapes visually but outline/border still follow box model */ clip-path: polygon(0% 0%, 80% 0%, 100% 50%, 80% 100%, 0% 100%); /* Note: border, box-shadow, outline do NOT follow clip-path shape */ } } /* With border-shape — border, shadow, and outline follow the shape */ @supports (border-shape: shape(from 0% 0%, line to 100% 0%, close)) { .shaped-el { border-shape: shape(from 0% 0%, line to 80% 0%, line to 100% 50%, line to 80% 100%, line to 0% 100%, close); border: 2px solid #000; /* follows the shape */ box-shadow: 4px 4px 0 #000; /* follows the shape */ } } /* JavaScript feature detection */ const SUPPORTS_BORDER_SHAPE = CSS.supports('border-shape', 'shape(from 0% 0%, line to 100% 0%, close)');
