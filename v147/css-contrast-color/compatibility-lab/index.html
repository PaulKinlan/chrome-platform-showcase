Fallback pattern

/* Detect contrast-color() support */ const SUPPORTS_CONTRAST_COLOR = CSS.supports('color', 'contrast-color(red)'); /* CSS-only: use contrast-color() when available */ @supports (color: contrast-color(red)) { .dynamic-text { color: contrast-color(var(--bg)); } } /* @supports not guard: JS luminance fallback */ @supports not (color: contrast-color(red)) { /* JS adds .dark-bg or .light-bg class based on luminance */ .dark-bg .dynamic-text { color: #ffffff; } .light-bg .dynamic-text { color: #000000; } } /* JavaScript luminance fallback */ function contrastColor(hexColor) { if (SUPPORTS_CONTRAST_COLOR) return `contrast-color(${hexColor})`; const hex = hexColor.replace('#', ''); const r = parseInt(hex.slice(0, 2), 16) / 255; const g = parseInt(hex.slice(2, 4), 16) / 255; const b = parseInt(hex.slice(4, 6), 16) / 255; const lin = c => c <= 0.04045 ? c/12.92 : Math.pow((c+0.055)/1.055, 2.4); const L = 0.2126 * lin(r) + 0.7152 * lin(g) + 0.0722 * lin(b); return L > 0.179 ? '#000000' : '#ffffff'; } // Apply to all dynamic backgrounds: document.querySelectorAll('[data-bg]').forEach(el => { el.style.color = contrastColor(el.dataset.bg); });
