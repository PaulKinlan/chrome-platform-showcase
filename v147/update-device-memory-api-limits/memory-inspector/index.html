// The API is unchanged — just the possible return values
const mem = navigator.deviceMemory; // number | undefined

// Chrome <147 possible values: 0.25, 0.5, 1, 2, 4, 8
// Chrome 147+ possible values:
//   Android:   2, 4, 8
//   Desktop:   2, 4, 8, 16, 32

// Typical usage — tiered content loading:
if (mem >= 8) {
  loadHighFidelityAssets();
} else if (mem >= 4) {
  loadMediumFidelityAssets();
} else {
  loadLowFidelityAssets();
}
