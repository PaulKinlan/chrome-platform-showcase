Fallback pattern

/* Detect SoftNavigation entry type */ const HAS_SOFT_NAV = 'PerformanceObserver' in window && PerformanceObserver.supportedEntryTypes?.includes('soft-navigation'); /* Setup observer */ if (HAS_SOFT_NAV) { new PerformanceObserver((list) => { for (const entry of list.getEntries()) { console.log('soft-nav:', entry.name, entry.startTime, entry.duration); // Send to RUM analytics… } }).observe({ type: 'soft-navigation', buffered: true }); } else { // Fallback: intercept history.pushState / popstate const origPushState = history.pushState.bind(history); history.pushState = function (...args) { performance.mark('soft-nav-start-' + args[2]); origPushState(...args); }; window.addEventListener('popstate', (e) => { performance.mark('soft-nav-start-' + location.pathname); }); }
