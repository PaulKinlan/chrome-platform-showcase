Fallback pattern

/* Native: text-overflow clip on focus */ .truncated { overflow: hidden; text-overflow: ellipsis; white-space: nowrap; } .truncated:focus { overflow: auto; text-overflow: clip; } /* @supports guard for older browsers */ @supports not selector(.truncated:focus) { /* JS adds/removes a class instead */ .truncated.revealed { overflow: auto; text-overflow: clip; } } /* JavaScript fallback */ function applyClipTextFallback(el) { // Test if :focus text-overflow is supported const testEl = document.createElement('input'); testEl.style.overflow = 'hidden'; document.body.appendChild(testEl); testEl.focus(); const focused = getComputedStyle(testEl).overflow; document.body.removeChild(testEl); const nativeSupported = focused === 'hidden'; // would be 'auto' if native kicks in if (!nativeSupported) { el.addEventListener('focus', () => el.classList.add('revealed')); el.addEventListener('blur', () => el.classList.remove('revealed')); } } document.querySelectorAll('.truncated').forEach(applyClipTextFallback);
