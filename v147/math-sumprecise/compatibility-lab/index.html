/* Detect Math.sumPrecise */ const HAS_SUM_PRECISE = typeof Math.sumPrecise === 'function'; /* Kahan compensated summation fallback */ function kahanSum(values) { let sum = 0, comp = 0; for (const v of values) { const y = v - comp; const t = sum + y; comp = (t - sum) - y; sum = t; } return sum; } /* Unified API */ function sumPrecise(values) { if (HAS_SUM_PRECISE) return Math.sumPrecise(values); return kahanSum(values); } /* Usage */ const total = sumPrecise([0.1, 0.2, 0.3]); console.log(total); // 0.6 (precise) not 0.6000000000000001
