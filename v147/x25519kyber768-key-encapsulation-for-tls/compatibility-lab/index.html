Context notes

/* X25519Kyber768 is a TLS extension — not directly visible in JS */ /* The hybrid key exchange happens automatically in the TLS handshake. Chrome 147 negotiates X25519Kyber768 when both client and server support it. There is no JavaScript API to observe or control which TLS KEM is used. */ /* What you CAN probe from JS: */ // 1. WebCrypto X25519 support (the classical component) async function hasX25519() { try { await crypto.subtle.generateKey({ name: 'X25519' }, false, ['deriveKey']); return true; } catch { return false; } } // 2. WebCrypto ECDH support (alternative classical KEM) async function hasECDH() { try { await crypto.subtle.generateKey({ name: 'ECDH', namedCurve: 'P-256' }, false, ['deriveKey']); return true; } catch { return false; } } // 3. Check if connection is TLS 1.3 (required for hybrid KEMs) // No JS API for this — infer from Performance ResourceTiming nextHopProtocol function getTLSVersion() { const entries = performance.getEntriesByType('navigation'); if (entries.length && entries[0].nextHopProtocol) { return entries[0].nextHopProtocol; // 'h2', 'h3', etc. } return 'unknown'; } /* Note: X25519Kyber768 is transparent to web developers. If you need post-quantum security for application-layer data, use WebCrypto's ML-KEM once it's exposed there. For TLS, Chrome 147 handles it automatically. */
