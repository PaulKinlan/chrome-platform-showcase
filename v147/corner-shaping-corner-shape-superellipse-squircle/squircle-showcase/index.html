/* The iOS-style squircle in 2 lines of CSS */
.app-icon {
  border-radius: 22%;
  corner-shape: squircle;
}

/* Progressive enhancement fallback */
@supports not (corner-shape: squircle) {
  .app-icon {
    border-radius: 22%; /* rounded rect — acceptable fallback */
  }
}
