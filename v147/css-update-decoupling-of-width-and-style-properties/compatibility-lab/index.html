Fallback pattern

/* Detect Chrome 147 width/style decoupling */ function getBorderWidthFixed() { const el = document.createElement('div'); el.style.cssText = 'position:absolute;left:-9999px;border-style:none;border-width:4px;'; document.body.appendChild(el); const computed = getComputedStyle(el).borderTopWidth; document.body.removeChild(el); // Chrome 147+: returns "4px"; older Chrome: returns "0px" return computed !== '0px'; } const HAS_DECOUPLED_WIDTH = getBorderWidthFixed(); /* Safe pattern: read border-width without style dependency */ function getEffectiveBorderWidth(el) { const cs = getComputedStyle(el); if (HAS_DECOUPLED_WIDTH) { // Chrome 147+ / Firefox / Safari — computed value is author value return parseFloat(cs.borderTopWidth); } // Old Chrome: computed value may be 0 if border-style is none/hidden // Workaround: temporarily set a visible border-style to read the width const origStyle = el.style.borderStyle; el.style.borderStyle = 'solid'; const width = parseFloat(getComputedStyle(el).borderTopWidth); el.style.borderStyle = origStyle; return width; }
