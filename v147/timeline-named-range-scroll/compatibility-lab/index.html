Fallback pattern

/* Detect scroll-range support via CSSStyleSheet injection */ const SUPPORTS_SCROLL_RANGE = (() => { try { const ss = new CSSStyleSheet(); ss.replaceSync('.x { animation-range: scroll; }'); return ss.cssRules[0]?.style?.animationRange !== undefined; } catch { return false; } })(); /* CSS: use animation-range: scroll when supported */ @supports (animation-range: scroll) { .scroll-progress { animation: grow-bar linear; animation-timeline: scroll(nearest block); animation-range: scroll; width: 100%; /* animation from 0→100% via transform:scaleX is typical */ } } /* JavaScript fallback for non-supporting browsers */ if (!SUPPORTS_SCROLL_RANGE) { const bar = document.querySelector('.scroll-progress'); const scroller = document.querySelector('.scroll-container'); scroller.addEventListener('scroll', () => { const { scrollTop, scrollHeight, clientHeight } = scroller; const pct = (scrollTop / (scrollHeight - clientHeight)) * 100; bar.style.width = pct + '%'; }, { passive: true }); }
