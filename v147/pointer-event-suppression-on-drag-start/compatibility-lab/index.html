    Fallback pattern

    /* Detect whether pointercancel fires on drag start */ function setupDragElement(el) { let isPointerDown = false; let isDragging = false; let sawDragStartCancel = false; el.addEventListener('pointerdown', (event) => { isPointerDown = true; isDragging = false; sawDragStartCancel = false; el.setPointerCapture?.(event.pointerId); }); el.addEventListener('dragstart', () => { isDragging = true; }); // Chrome 147+: this fires after dragstart — guaranteed cleanup point. el.addEventListener('pointercancel', () => { sawDragStartCancel = isDragging; isPointerDown = false; resetCursor(el); }); // Fallback: older browsers need dragend cleanup because no pointercancel arrives. el.addEventListener('dragend', () => { if (!sawDragStartCancel) { isPointerDown = false; resetCursor(el); } isDragging = false; }); } function resetCursor(el) { el.style.cursor = ''; el.classList.remove('dragging'); }
