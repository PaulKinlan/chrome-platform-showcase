Fallback pattern

/* Detect whether pointercancel fires on drag start */ const HAS_SUPPRESS = (() => { // Chrome 147+ sends pointercancel after dragstart; older Chrome does not. // We can't feature-detect this statically — use the live drag test above // to confirm at runtime, then cache the result. return 'PointerEvent' in window; // base check; full detect requires a drag })(); /* Safe pointer-state guard pattern for drag libraries */ function setupDragElement(el) { let isPointerDown = false; let isDragging = false; el.addEventListener('pointerdown', () => { isPointerDown = true; isDragging = false; }); el.addEventListener('dragstart', () => { isDragging = true; // Chrome 147+: pointercancel follows automatically — don't reset here. // Older Chrome: no pointercancel will arrive — reset now as a fallback. if (!HAS_SUPPRESS) { isPointerDown = false; el.releasePointerCapture(/* pointerId from pointerdown event */); } }); // Chrome 147+: this fires after dragstart — guaranteed cleanup point. el.addEventListener('pointercancel', () => { isPointerDown = false; resetCursor(el); }); // Fallback: also reset on dragend for older browsers el.addEventListener('dragend', () => { isPointerDown = false; resetCursor(el); }); } function resetCursor(el) { el.style.cursor = ''; el.classList.remove('dragging'); }
