how it works

const el = document.querySelector('[draggable]');

// These three now fire on drag start in Chrome 147+
el.addEventListener('pointercancel', (e) => {
  // "I've lost control — a drag captured the pointer"
  console.log('pointercancel', e.pointerId);
  resetDragState(); // ← reliable cleanup, no heuristics needed
});
el.addEventListener('pointerout',   () => console.log('pointerout'));
el.addEventListener('pointerleave', () => console.log('pointerleave'));
