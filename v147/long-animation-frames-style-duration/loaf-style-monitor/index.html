    how it works

    const observer = new PerformanceObserver(list => {
  for (const entry of list.getEntries()) {
    // New fields in Chrome 147 (origin trial)
    console.log("styleDuration:",        entry.styleDuration);
    console.log("forcedStyleDuration:",  entry.forcedStyleDuration);
    console.log("layoutDuration:",       entry.layoutDuration);
    console.log("forcedLayoutDuration:", entry.forcedLayoutDuration);

    // Existing field — total frame time
    console.log("duration:",             entry.duration);
  }
});

observer.observe({ type: "long-animation-frame", buffered: true });
