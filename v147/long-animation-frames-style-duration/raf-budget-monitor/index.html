the new fields

new PerformanceObserver(list => {
  for (const entry of list.getEntries()) {
    // Chrome 147+ new fields:
    console.log({
      duration:             entry.duration,           // total frame ms
      styleDuration:        entry.styleDuration,       // CSS recalc time
      forcedStyleDuration:  entry.forcedStyleDuration, // JS-triggered recalcs
      layoutDuration:       entry.layoutDuration,      // layout/reflow time
      forcedLayoutDuration: entry.forcedLayoutDuration,// JS-triggered layout
    });
  }
}).observe({ type: 'long-animation-frame', buffered: true });
