Probes long-animation-frame in PerformanceObserver.supportedEntryTypes and checks whether LoAF entries expose styleDuration, forcedStyleDuration, layoutDuration, and forcedLayoutDuration. Triggers a real slow frame to capture a live LoAF entry and inspect all its fields. Provides a performance.measure() fallback.

Origin trial. The new styleDuration / layoutDuration fields require the LongAnimationFrameStyleDuration origin trial or chrome://flags/#enable-experimental-web-platform-features. Basic LoAF (long-animation-frame) is available in Chrome 123+ without a flag.

Fallback pattern

/* LoAF + styleDuration detection */ const HAS_LOAF = 'PerformanceObserver' in window && PerformanceObserver.supportedEntryTypes?.includes('long-animation-frame'); /* Check for Chrome 147 style-duration fields */ let HAS_STYLE_DURATION = false; function observeLoAF(callback) { if (!HAS_LOAF) return null; const observer = new PerformanceObserver((list) => { for (const entry of list.getEntries()) { // Detect styleDuration on first entry if (!HAS_STYLE_DURATION && 'styleDuration' in entry) { HAS_STYLE_DURATION = true; } const data = { duration: entry.duration, renderStart: entry.renderStart, // Chrome 123+ fields always present styleAndLayoutStart: entry.styleAndLayoutStart, // Chrome 147+ OT fields — undefined in older Chrome styleDuration: entry.styleDuration, layoutDuration: entry.layoutDuration, forcedStyleDuration: entry.forcedStyleDuration, forcedLayoutDuration: entry.forcedLayoutDuration, }; callback(data); } }); observer.observe({ type: 'long-animation-frame', buffered: true }); return observer; } /* Fallback: manual frame timing via performance.mark() */ function measureFrame(label, fn) { performance.mark(label + '-start'); fn(); performance.mark(label + '-end'); performance.measure(label, label + '-start', label + '-end'); return performance.getEntriesByName(label, 'measure')[0]?.duration ?? 0; }
