Fallback pattern

/* Check which pseudo types return a CSSPseudoElement object */ function checkPseudoSupport(el, type) { if (typeof el.pseudo !== 'function') return 'no-api'; const obj = el.pseudo(type); if (obj == null) return 'null'; if (typeof obj.getBoundingClientRect === 'function') return 'full'; return 'partial'; } /* Safe pseudo-element style reader */ function getPseudoComputedStyle(el, pseudo) { if (typeof el.pseudo === 'function') { const pseudoEl = el.pseudo(pseudo); if (pseudoEl) return getComputedStyle(pseudoEl); } // Fallback: legacy two-arg form (works for all pseudo types) return getComputedStyle(el, pseudo); } /* Usage */ const el = document.querySelector('.my-element'); const beforeStyle = getPseudoComputedStyle(el, '::before'); console.log(beforeStyle.content, beforeStyle.color); // ::backdrop — use getComputedStyle fallback (not a CSSPseudoElement in Chrome 147) const dialogEl = document.querySelector('dialog'); const bdStyle = getComputedStyle(dialogEl, '::backdrop'); console.log(bdStyle.backgroundColor);
