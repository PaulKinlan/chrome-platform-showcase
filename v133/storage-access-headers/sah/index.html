the headers

Sec-Fetch-Storage-Access: active

// Server -> browser:
Activate-Storage-Access: load   // grant storage on the loaded resource
// or
Activate-Storage-Access: retry; allowed-origin="https://a.example"
// or
Vary: Sec-Fetch-Storage-Access
