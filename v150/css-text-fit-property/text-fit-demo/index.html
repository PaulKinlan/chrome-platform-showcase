code

/* Chrome 150+ */
.responsive-heading {
  text-fit: contain;          /* expand/shrink to fill container */
  min-font-size: 1rem;        /* never smaller than 1rem */
  max-font-size: 5rem;        /* never larger than 5rem */
}

/* Graceful fallback for older browsers */
@supports not (text-fit: contain) {
  .responsive-heading {
    font-size: clamp(1rem, 5vw, 5rem); /* approximate */
  }
}

/* scale-down: only shrink, never grow beyond font-size value */
.label {
  font-size: 1.5rem;
  text-fit: scale-down;       /* stays 1.5rem if it fits; shrinks if not */
}
