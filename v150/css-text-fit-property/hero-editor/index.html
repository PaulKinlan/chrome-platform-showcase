Try: 
/* One rule — works for any text length, any container width */
.hero-headline {
  font-size: 5rem;     /* initial / max size */
  text-fit: contain;
  min-font-size: 1rem; /* never smaller than 1rem */
  max-font-size: 8rem; /* never larger than 8rem */
  white-space: nowrap; /* keeps text on one line */
}

/* Only shrink, never grow — good for fixed-height containers */
.card-title {
  font-size: 1.5rem;
  text-fit: scale-down;
}

/* Graceful fallback for non-supporting browsers */
@supports not (text-fit: contain) {
  .hero-headline { font-size: clamp(1rem, 8vw, 5rem); }
}
