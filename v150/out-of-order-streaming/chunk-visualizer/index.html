// Reading a stream — API is unchanged, delivery improved
const response = await fetch('/api/stream');
const reader = response.body.getReader();

while (true) {
  const { done, value } = await reader.read();
  if (done) break;
  // Chrome 150+: dispatched immediately when available,
  // even if an earlier chunk is still in-flight.
  process(value);
}
