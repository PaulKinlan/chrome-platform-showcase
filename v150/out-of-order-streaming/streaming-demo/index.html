Streaming Demo

Consumes a ReadableStream chunk by chunk, showing each chunk's arrival time and size. Chrome 150 dispatches available chunks immediately without waiting for earlier chunks that may be delayed.

This demo creates a local ReadableStream with simulated chunk delays to show how the stream reader receives chunks. In a real network scenario, Chrome 150 removes in-browser head-of-line blocking so network chunks are dispatched as soon as they arrive.

// Creating a ReadableStream with staggered chunk delivery
function makeStream(chunks, delays) {
  let i = 0;
  return new ReadableStream({
    pull(controller) {
      if (i >= chunks.length) { controller.close(); return; }
      return new Promise(resolve => {
        setTimeout(() => {
          controller.enqueue(new TextEncoder().encode(chunks[i]));
          i++;
          resolve();
        }, delays[i - 1] || 0);
      });
    }
  });
}

const reader = makeStream(data, delays).getReader();
const decoder = new TextDecoder();

while (true) {
  const { done, value } = await reader.read();
  if (done) break;
  // Chrome 150+: dispatched immediately as chunks arrive
  display.textContent += decoder.decode(value, { stream: true });
}
