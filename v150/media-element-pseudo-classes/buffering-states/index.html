Active rule: video:buffering { ... } 
/* :buffering — media is playing but waiting for more data */
video:buffering {
  outline: 2px solid var(--accent-amber);
}
video:buffering::before {
  content: '';
  /* show loading spinner overlay */
}

/* :seeking — user has requested a position change */
video:seeking {
  opacity: 0.7;
}

/* :stalled — browser cannot fetch data */
video:stalled {
  border-color: var(--accent-rose);
}

/* All six media pseudo-classes:
   :playing    — media is actively playing
   :paused     — media is paused or not yet played
   :seeking    — seeking to a new position
   :buffering  — playing but waiting for data
   :stalled    — cannot retrieve data (network error)
   :muted      — audio is muted                    */
