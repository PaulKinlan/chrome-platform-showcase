code

<!-- HTML attribute -->
<textarea autocorrect="on"></textarea>   <!-- OS corrects typos -->
<textarea autocorrect="off"></textarea>  <!-- Raw input preserved -->

<!-- Use cases for autocorrect="off":
  - Code editors / technical input fields
  - Username / email / URL fields
  - Search boxes where misspellings are intentional
  - Template variables like {{user.name}}
-->

<!-- Use cases for autocorrect="on" (default):
  - Comment boxes
  - Long-form text areas
  - Chat / messaging fields
-->

// JavaScript API (Chrome 150+)
const input = document.querySelector('input');
console.log(input.autocorrect);     // "on" or "off"
input.autocorrect = 'off';          // programmatic control
