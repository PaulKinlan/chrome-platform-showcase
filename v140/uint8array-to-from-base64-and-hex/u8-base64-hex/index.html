const enc = new TextEncoder().encode("hello world");
enc.toBase64();              // "aGVsbG8gd29ybGQ="
enc.toBase64({ alphabet: "base64url" });
enc.toHex();                 // "68656c6c6f20776f726c64"
Uint8Array.fromBase64("aGVsbG8=");
Uint8Array.fromHex("68656c6c6f");
