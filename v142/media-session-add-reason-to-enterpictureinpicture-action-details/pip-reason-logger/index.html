relevant API

navigator.mediaSession.setActionHandler("enterpictureinpicture", (details) => {
  // details.reason values (Chrome 142+):
  //   "user-initiated" — user clicked the PiP button
  //   "auto"           — browser auto-triggered (tab hidden, etc.)
  //   "media-session"  — triggered via media session action
  const reason = details?.reason ?? "unknown";
  console.log("PiP reason:", reason);
});
