what changed in Chrome 142

// Same-origin renderer-initiated navigations now preserve
// navigator.userActivation.hasBeenActive = true
//
// Before 142: hasBeenActive reset on every navigation
// After  142: hasBeenActive persists across same-origin
//             renderer-initiated navigations (pushState,
//             replaceState, same-origin link clicks)

// Check it:
window.addEventListener("popstate", () => {
  console.log(navigator.userActivation.hasBeenActive);
  // true in Chrome 142+ if user clicked before navigating
});
