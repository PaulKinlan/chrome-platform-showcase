Handler code

navigation.addEventListener('navigate', (event) => { event.intercept({ handler: () => Promise.resolve() }); // SPA intercept const el = event.sourceElement; // Chrome 134: the triggering element if (el === null) { analytics.track('programmatic_nav', { url: event.destination.url }); return; } if (el.tagName === 'A') { analytics.track('link_click', { id: el.id, intent: el.dataset.intent, category: el.dataset.category, href: el.href, text: el.textContent.trim(), }); } else if (el.tagName === 'FORM') { analytics.track('form_submit', { id: el.id, intent: el.dataset.intent, action: el.action, data: Object.fromEntries(new FormData(el)), }); } });
