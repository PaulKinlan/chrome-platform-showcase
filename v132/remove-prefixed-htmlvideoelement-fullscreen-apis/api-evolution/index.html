# rewrites:
# webkitRequestFullscreen   -> requestFullscreen
# webkitExitFullscreen      -> exitFullscreen     (and switch to document.)
# webkitDisplayingFullscreen -> document.fullscreenElement === el
# webkitSupportsFullscreen  -> document.fullscreenEnabled
# webkitbeginfullscreen     -> fullscreenchange (document, check enter)
# webkitendfullscreen       -> fullscreenchange (document, check exit)
