API availability in this browser

// Standard API — works in all modern browsers
const video = document.querySelector('video');

// Enter fullscreen
await video.requestFullscreen();           // standard ✓

// Exit
document.exitFullscreen();                 // standard ✓

// Events (standard)
document.addEventListener('fullscreenchange', handler);
document.addEventListener('fullscreenerror',  handler);
video.addEventListener('fullscreenchange', handler);

// ─── REMOVED in Chrome 132 ───────────────────────────
video.webkitEnterFullscreen();             // ← throws TypeError
video.webkitExitFullscreen();              // ← throws TypeError
video.webkitRequestFullscreen();           // ← throws TypeError
video.addEventListener('webkitfullscreenchange', …); // ← never fires

// ─── Migration pattern ───────────────────────────────
const enterFS = (el) => {
  if (el.requestFullscreen) return el.requestFullscreen();
  // No more webkit fallback needed — Chrome 71+ has standard API
};
