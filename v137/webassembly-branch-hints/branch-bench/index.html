the wat

(module
  (func (export "loop") (param $n i32) (result i32)
    (local $sum i32)
    (block $exit
      (loop $body
        (br_if $exit (i32.eqz (local.get $n)))      ;; (@metadata.code.branch_hint "\00") <-- unlikely
        (local.set $sum (i32.add (local.get $sum) (local.get $n)))
        (local.set $n   (i32.sub (local.get $n) (i32.const 1)))
        (br $body)))
    (local.get $sum)))
