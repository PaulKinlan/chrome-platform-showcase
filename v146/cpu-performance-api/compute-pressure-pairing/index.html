Compute Pressure Pairing

The CPU Performance API gives a static device capability tier (low / mid / high). The Compute Pressure API gives a live CPU utilisation state (nominal → fair → serious → critical). Together they let apps set a quality ceiling at startup and then adapt within it in real time — static tier constrains the ceiling, live pressure controls the floor.

// Static tier — read once at startup
async function getDeviceTier() {
  const concurrency = navigator.hardwareConcurrency ?? 2;
  const memory      = navigator.deviceMemory        ?? 1;
  if      (concurrency >= 8  && memory >= 8)  return 'high';
  else if (concurrency >= 4  && memory >= 4)  return 'mid';
  else                                        return 'low';
}

// Live pressure — subscribe for the session
const pressureObs = new PressureObserver(records => {
  const { state } = records.at(-1); // 'nominal' | 'fair' | 'serious' | 'critical'
  applyQualityPreset(currentTier, state);
});
await pressureObs.observe('cpu', { sampleInterval: 1000 });

function applyQualityPreset(tier, pressure) {
  // Ceiling set by tier; floor pushed down by pressure
  const preset = QUALITY_MATRIX[tier][pressure];
  videoSender.setParameters({ maxBitrate: preset.bitrate });
  particleSystem.setBudget(preset.particles);
  if (!preset.mlEnabled) mlModel.suspend();
}
