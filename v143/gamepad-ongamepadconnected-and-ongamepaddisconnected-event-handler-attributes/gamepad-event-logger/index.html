Code comparison

active 
// Works in all browsers
window.addEventListener(
  'gamepadconnected',
  (e) => {
    console.log(e.gamepad);
  }
);
window.addEventListener(
  'gamepaddisconnected',
  (e) => {
    console.log(e.gamepad);
  }
);
active 
// New in Chrome 143
window
  .ongamepadconnected
  = (e) => {
    console.log(e.gamepad);
  };
window
  .ongamepaddisconnected
  = (e) => {
    console.log(e.gamepad);
  };
