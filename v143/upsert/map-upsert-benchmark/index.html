// Pattern 1: Classic has/get/set const map = new Map(); for (const word of words) { if (map.has(word)) { map.set(word, map.get(word) + 1); } else { map.set(word, 1); } }
// Pattern 2: Get-or-default (nullish coalescing) const map = new Map(); for (const word of words) { map.set(word, (map.get(word) ?? 0) + 1); }
// Pattern 3: Map.prototype.upsert() — Chrome 143 // upsert(key, updateFn, insertFn) // updateFn: called with existing value to produce new value // insertFn: called with no args when key is absent const map = new Map(); for (const word of words) { map.upsert(word, v => v + 1, () => 1); } // Polyfill used on browsers without native upsert: if (typeof Map.prototype.upsert !== 'function') { Map.prototype.upsert = function(key, updateFn, insertFn) { if (this.has(key)) { this.set(key, updateFn(this.get(key))); } else { this.set(key, insertFn()); } return this.get(key); }; }
