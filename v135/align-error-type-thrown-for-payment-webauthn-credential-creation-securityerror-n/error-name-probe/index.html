recommended catch block

try {
  await navigator.credentials.create({ publicKey, payment: { isPayment: true } });
} catch (err) {
  switch (err.name) {
    case "SecurityError":   showSecurityHint();   break;  // 135+
    case "NotAllowedError": showCancelHint();     break;
    case "AbortError":      showAbortedHint();    break;
    default:                report(err);
  }
}
