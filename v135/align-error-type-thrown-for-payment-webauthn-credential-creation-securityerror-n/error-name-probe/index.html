recommended catch block

try {
  await navigator.credentials.create({ publicKey });
} catch (error) {
  switch (error.name) {
    case "NotAllowedError":
      showActivationOrCancelHint();
      break;
    case "SecurityError":
      showPolicyOrOriginFailure();
      break;
    case "AbortError":
      showAbortedHint();
      break;
    default:
      report(error);
  }
}
