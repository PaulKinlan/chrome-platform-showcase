the code

const a = new Float16Array([0.1, 0.2, 0.3]);
console.log(a[0]);                       // 0.0999755859375
console.log(a.byteLength / a.length);    // 2

const dv = new DataView(a.buffer);
console.log(dv.getUint16(0, true).toString(16)); // raw bits
