/* ::column styles the grid column track directly — no wrapper divs */
.kanban-board::column {
  border-right: 2px solid var(--border-black);
}

/* Nth-column targeting */
.kanban-board.colored::column:nth-child(1) { background: #fff7ed; }
.kanban-board.colored::column:nth-child(2) { background: #eff6ff; }
.kanban-board.colored::column:nth-child(3) { background: #f0fdf4; }
.kanban-board.colored::column:nth-child(4) { background: #fdf4ff; }

/* Before ::column, you needed an extra div per column:
   <div class="col-wrapper col-1">
     <div class="col-header">…</div>
     <div class="col-items">…</div>
   </div> */
