// Chrome 130: improved IndexedDB error reporting
// DOMException.name is now meaningful for large-value failures

request.onerror = (e) => {
  const err = e.target.error;

  switch (err.name) {
    case 'DataError':
      // Data couldn't be serialised / deserialised
      // → clear the offending key, inform user
      break;
    case 'UnknownError':
      // OS-level I/O failure (blob file deleted, disk full)
      // → retry once; if persistent, prompt re-upload
      break;
    case 'QuotaExceededError':
      // Storage quota hit
      // → call navigator.storage.estimate(), offer cleanup UI
      break;
    default:
      // Unexpected error — log to server
      reportToServer(err.name, err.message);
  }
};
