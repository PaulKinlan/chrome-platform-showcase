Credential Audit

Chrome 130 adds PublicKeyCredential.signalUnknownCredential(), signalCurrentUserDetails(), and signalAllAcceptedCredentials(). This audit tool simulates a relying party's credential store — with synced, stale, and revoked entries — and calls the correct Signal API method for each state, letting you see exactly which signals to emit and when.

// Chrome 130: WebAuthn Signal API — emit signals after login
async function auditCredentials(currentUser, serverCredIds) {
  // 1. Signal all credentials the server still accepts
  //    → removes stale entries from the OS passkey picker
  await PublicKeyCredential.signalAllAcceptedCredentials({
    rpId: 'example.com',
    userId: currentUser.userId,      // base64url
    allAcceptedCredentialIds: serverCredIds  // base64url[]
  });

  // 2. Keep username/displayName in sync with the server
  await PublicKeyCredential.signalCurrentUserDetails({
    rpId: 'example.com',
    userId: currentUser.userId,
    name: currentUser.username,
    displayName: currentUser.displayName,
  });
}

// After account deletion / credential revocation:
async function revokeCredential(credId) {
  await PublicKeyCredential.signalUnknownCredential({
    rpId: 'example.com',
    credentialId: credId,  // base64url
  });
}
