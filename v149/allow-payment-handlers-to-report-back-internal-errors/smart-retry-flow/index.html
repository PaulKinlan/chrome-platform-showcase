Smart Retry Flow

A simulated checkout that reacts differently to each error type: a user cancellation gets a gentle nudge to continue shopping, while an internal app error triggers an automatic retry with a fallback payment method.

// merchant page — Chrome 149+
async function pay(method) {
  const request = new PaymentRequest([{ supportedMethods: method }], details);
  try {
    const response = await request.show();
    await response.complete('success');
    showOrderConfirmation(); // payment worked

  } catch (err) {
    if (err.name === 'AbortError') {
      // User pressed cancel — respect it, don't retry
      showCancelMessage('Your basket is saved for later.');

    } else if (err.name === 'OperationError') {
      // Payment app bug — user didn't cancel, safe to retry
      await pay(FALLBACK_METHOD); // switch to bank transfer

    } else {
      showGenericError(err.message);
    }
  }
}
