Code comparison

Before — JavaScript workaround (12 lines)

const el = document.querySelector('.truncated');

el.addEventListener('focus', () => {
  // Remove ellipsis manually
  el.style.textOverflow = 'clip';
  el.style.overflowX = 'auto';
});

el.addEventListener('blur', () => {
  // Restore ellipsis
  el.style.textOverflow = 'ellipsis';
  el.style.overflowX = 'hidden';
  el.scrollLeft = 0;
});

Chrome 149 — CSS only (0 extra lines)

/* Just set these on the element */
.cell {
  white-space: nowrap;
  overflow: hidden;
  text-overflow: ellipsis;
}

/* That's it. Chrome 149 handles the
   focus/blur reveal automatically for
   contenteditable and input elements.
   No JavaScript. */
