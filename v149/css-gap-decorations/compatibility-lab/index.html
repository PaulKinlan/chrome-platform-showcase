Progressive enhancement pattern

/* Fallback: border on cells (works everywhere) */ .grid-cell { border-right: 2px solid var(--border-black); border-bottom: 2px solid var(--border-black); } .grid-cell:last-child { border-right: none; } .grid-cell:nth-last-child(-n+3) { border-bottom: none; } /* Enhancement: gap decorations (Chrome 149 / Firefox) */ @supports (column-rule: 1px solid) and (row-rule: 1px solid) { /* Remove border hacks — gap decorations take over */ .grid-cell { border: none; } .grid { column-rule: 2px solid var(--border-black); row-rule: 2px solid var(--border-black); } } /* Feature detection in JavaScript */ const HAS_GAP_DECORATIONS = CSS.supports('column-rule', '1px solid') && CSS.supports('row-rule', '1px solid');
