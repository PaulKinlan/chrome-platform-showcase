<script>
// Monitor state changes
const permEl = document.querySelector('permission');
permEl.addEventListener('promptdismiss', () => {
  navigator.permissions.query({ name: 'camera' })
    .then(s => console.log('camera:', s.state));
});

// Compare: classic getUserMedia approach
async function requestViaGUM() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({
      video: true, audio: true
    });
    videoEl.srcObject = stream;
  } catch (err) {
    if (err.name === 'NotAllowedError') showDeniedFallback();
  }
}
</script>
