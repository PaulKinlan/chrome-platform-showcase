// Benchmarking the Language Detector API async function benchmark(texts) { const detector = await self.ai.languageDetector.create(); const results = []; for (const { text, expected } of texts) { const t0 = performance.now(); const [top] = await detector.detect(text); const latency = performance.now() - t0; results.push({ detected: top.detectedLanguage, confidence: top.confidence, correct: top.detectedLanguage === expected, latencyMs: latency.toFixed(1), }); } return results; }
