// Defensive Translator usage async function safeTranslate(text, from, to) { if (!window.Translator) return { error: 'not-supported' }; if (!text?.trim()) return { error: 'empty-input' }; if (from === to) return { result: text }; // no-op const avail = await Translator.availability({ sourceLanguage: from, targetLanguage: to }); if (avail === 'no') return { error: 'pair-unavailable' }; try { const t = await Translator.create({ sourceLanguage: from, targetLanguage: to }); return { result: await t.translate(text) }; } catch (e) { return { error: 'inference-error', message: e.message }; } }
