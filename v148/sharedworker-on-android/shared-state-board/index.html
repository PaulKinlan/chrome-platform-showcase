The SharedWorker pattern

// In every tab (page.js):
const worker = new SharedWorker('./worker.js');
worker.port.addEventListener('message', (e) => {
  const { type, ...payload } = e.data;
  switch (type) {
    case 'COUNTER': updateCounter(payload.counter); break;
    case 'NEW_MSG':  appendMessage(payload.msg); break;
    case 'USER_JOINED': updateUserList(payload.users); break;
    // …
  }
});
worker.port.start();

// Send events:
worker.port.postMessage({ type: 'INCREMENT' });
worker.port.postMessage({ type: 'SEND_MSG', text: 'Hello!' });

// In worker.js — all ports (tabs) share a single execution context:
const ports = new Map();
let counter = 0;
self.addEventListener('connect', (event) => {
  const port = event.ports[0];
  // … store port, handle messages, broadcast to all other ports
});

Chrome 148 enables this on Android — previously SharedWorker was silently unavailable on Chrome for Android. The fix aligns Android with desktop.
