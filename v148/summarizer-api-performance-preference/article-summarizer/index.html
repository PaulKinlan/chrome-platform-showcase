API usage

// Chrome 148: performance option on Summarizer.create()
const [fastSummarizer, qualitySummarizer] = await Promise.all([
  Summarizer.create({ performance: 'fast' }),
  Summarizer.create({ performance: 'quality' }),
]);

const [fast, quality] = await Promise.all([
  fastSummarizer.summarize(articleText),
  qualitySummarizer.summarize(articleText),
]);

// fast → lower latency, smaller model
// quality → richer output, larger model
