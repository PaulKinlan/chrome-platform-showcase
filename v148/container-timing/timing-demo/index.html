How to use the API


<section containertiming="hero-section">
  <img src="hero.jpg">
  <h1>Headline</h1>
</section>

// JS: observe with PerformanceObserver
const observer = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    console.log(entry.identifier);      // "hero-section"
    console.log(entry.startTime);       // ms from navigation
    console.log(entry.duration);        // 0 (instant event)
    console.log(entry.intersectionRect); // visible area
  }
});

observer.observe({ type: 'container-element', buffered: true });
